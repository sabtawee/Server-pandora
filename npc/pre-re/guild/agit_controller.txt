// WoE Start/Stop times
//======================================
-	script	Agit_Event	-1,{
	end;

OnClock2100:	//start time for
OnClock2200:	//end time for 


OnAgitInit:
	// starting time checks
	if((gettime(4)==2) && (gettime(3)>=21 && gettime(3)<22) ||
	   (gettime(4)==6) && (gettime(3)>=21 && gettime(3)<22)) {
		if (!agitcheck()) {
			agitstart;
			announce "[ GuildWar ] : สงคราม Guild War ได้เริ่มต้นขึ้นแล้ว ณ บัดนี้ เปิดบ้าน พาย่อน4 กิฟเฟน1ลุย!!",0;
			callsub S_DisplayOwners;
		}
		end;
	}

	// end time checks
	if ((gettime(4)==2) && (gettime(3)==22) ||
	    (gettime(4)==6) && (gettime(3)==22)) {
 
		if (agitcheck()) {
			agitend;
			announce "[ GuildWar ] : สงคราม Guild War จบลงแล้ว  !!",0;
			callsub S_DisplayOwners;
		}
		end;
	}
	end;

S_DisplayOwners:
	setarray .@maps$[1],"gefg_cas01";
	setarray .@maps$[2],"payg_cas04";
	for( set .@i, 0; .@i <= 1; set .@i, .@i+1 ) {
		if (getcastledata(.@maps$[.@i],1))
			announce "The [" + getcastlename(.@maps$[.@i]) + "] Agit ได้ถูกยึดครองโดย Guild [" + getguildname(getcastledata(.@maps$[.@i],1)) + "]",bc_all|bc_woe;
		else
			announce "The [" + getcastlename(.@maps$[.@i]) + "] Agit ยังไม่มีผู้ใดครอบครอง",bc_all|bc_woe;
	}
	end;
}
